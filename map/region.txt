# A province can belong to several regions.

# random_new_world_region is used for RNW. Must be first in this list.
random_new_world_region = {
	discover_if = {
		capital_scope = { continent = europe }
		any_country = {
			capital_scope = { continent = europe }
			has_discovered_region = {
				region = FROM
				years = 50
			}
		}
		is_year = 1500
	}
	discover_if = {
		any_neighbor_country = {
			OR = {
				culture_group = ROOT
				religion_group = ROOT
			}
			has_discovered_region = {
				years = 75
				region = FROM
			}
		}
	}
	discover_if = {
		has_discovered_region = {
			region = FROM
			years = 20 #Num of years
		}
	}
	discover_if = {
		any_country = {
			capital_scope = { region = FROM }
			has_discovered_capital_region_of = { who = ROOT years = 30 # num of years
			}
		}
	}
}
