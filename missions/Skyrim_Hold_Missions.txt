usurp_falkreath = {
	
	type = country

	category = MIL
	ai_mission = yes
	
	allow = {
		normal_or_historical_nations = yes
		OR = {
			capital_scope = { area = north_falkreath_area }
			capital_scope = { area = south_falkreath_area }
			capital_scope = { area = sungard_area }
		}
		is_free_or_tributary_trigger = yes
		culture_group = nordic_g
		NOT = {
				OR = {
					tag = XSM
					tag = XEM
					tag = XFK
					tag = XHF
					tag = XHM
					tag = XPL
					tag = XRE
					tag = XRI
					tag = XWH
					tag = XWR
				}
			}
	}
	abort = {
		is_subject_other_than_tributary_trigger = yes
	}
	success = {
		tag = XFK
	}
	chance = {
		factor = 5000
	}
	immediate = {
		north_falkreath_area = {
			add_claim = ROOT
		}
		south_falkreath_area = {
			add_claim = ROOT
		}
		sungard_area = {
			add_claim = ROOT
		}
	}
	abort_effect = {

			if = {
				limit = {

					}
				}
